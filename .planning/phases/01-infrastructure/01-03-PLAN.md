    **frontend/src/stores/theme.ts:**
    ```typescript
    import { defineStore } from 'pinia'
    import { ref, watch } from 'vue'

    export const useThemeStore = defineStore('theme', () => {
      const prefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches
      const isDark = ref(localStorage.getItem('theme') === 'dark' || (localStorage.getItem('theme') === null && prefersDark))

      function toggle() {
        isDark.value = !isDark.value
        localStorage.setItem('theme', isDark.value ? 'dark' : 'light')
      }

      watch(isDark, (dark) => {
        document.documentElement.classList.toggle('dark', dark)
      }, { immediate: true })

      return { isDark, toggle }
    })
    ```

    **frontend/src/sw.ts:**
    Basic service worker that caches the app shell on install:
    ```typescript
    const CACHE_NAME = 'nomadbase-shell-v1'
    const SHELL_ASSETS = ['/', '/index.html']

    self.addEventListener('install', (event: ExtendableEvent) => {
      event.waitUntil(
        caches.open(CACHE_NAME).then((cache) => cache.addAll(SHELL_ASSETS))
      )
    })

    self.addEventListener('activate', (event: ExtendableEvent) => {
      event.waitUntil(
        caches.keys().then((names) =>
          Promise.all(names.filter((n) => n !== CACHE_NAME).map((n) => caches.delete(n)))
        )
      )
    })

    self.addEventListener('fetch', (event: FetchEvent) => {
      // Network-first for API, cache-first for shell
      if (event.request.url.includes('/api/')) return
      event.respondWith(
        caches.match(event.request).then((cached) => cached || fetch(event.request))
      )
    })
    ```
    Note: The service worker file needs to be copied/built to `/public/sw.js` or handled via vite plugin. Add a simple vite plugin or build step that copies sw.ts compiled output to the public directory.     **frontend/Dockerfile:**
    Multi-stage build: Node for building, nginx for serving:

    ```dockerfile
    # Build stage
    FROM node:20-alpine AS build
    WORKDIR /app
    COPY package.json package-lock.json ./
    RUN npm ci
    COPY . .
    RUN npm run build

    # Production stage
    FROM nginx:alpine
    COPY --from=build /app/dist /usr/share/nginx/html
    COPY nginx.conf /etc/nginx/conf.d/default.conf
    EXPOSE 80
    ```

    **frontend/nginx.conf:**
    ```nginx
    server {
        listen 80;
        server_name _;
        root /usr/share/nginx/html;
        index index.html;

        # SPA routing — all paths to index.html
        location / {
            try_files $uri $uri/ /index.html;
        }

        # Proxy API calls to FastAPI backend
        location /api/ {
            proxy_pass http://api:8000/api/;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
        }

        # Cache static assets
        location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff|woff2)$ {
            expires 1y;
            add_header Cache-Control "public, immutable";
        }

        # Serve manifest and SW without cache
        location = /manifest.json {
            add_header Cache-Control "no-cache";
        }
        location = /sw.js {
            add_header Cache-Control "no-cache";
        }
    }
    ```
