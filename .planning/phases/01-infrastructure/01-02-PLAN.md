    Create deterministic mapping from OSM tags to NomadBase place_type and lifestyle tags.

    ```python
    # OSM tag -> NomadBase place_type mapping
    # Priority order matters: first match wins
    PLACE_TYPE_RULES = [
        # (osm_key, osm_value, nomadbase_type)
        ("tourism", "attraction", "attraction"),
        ("tourism", "viewpoint", "viewpoint"),
        ("tourism", "museum", "cultural"),
        ("tourism", "artwork", "cultural"),
        ("tourism", "gallery", "cultural"),
        ("historic", "*", "cultural"),
        ("amenity", "restaurant", "restaurant"),
        ("amenity", "cafe", "cafe"),
        ("amenity", "fast_food", "restaurant"),
        ("amenity", "bar", "restaurant"),
        ("amenity", "pub", "restaurant"),
        ("leisure", "park", "park"),
        ("leisure", "garden", "park"),
        ("leisure", "nature_reserve", "park"),
        ("natural", "peak", "hiking"),
        ("natural", "waterfall", "hiking"),
        ("route", "hiking", "hiking"),
        ("highway", "path", None),  # skip generic paths
    ]

    # Lifestyle tag extraction
    LIFESTYLE_RULES = {
        "vegan": [("diet:vegan", "yes"), ("diet:vegan", "only")],
        "vegetarian": [("diet:vegetarian", "yes"), ("diet:vegetarian", "only")],
        "cheap": [("fee", "no"), ("access", "yes")],
        "nature": [("natural", "*"), ("leisure", "nature_reserve"), ("leisure", "park")],
        "hidden_gem": [],  # computed: low popularity + high rating signals (future)
    }

    def map_osm_tags(osm_tags: dict) -> tuple[str | None, dict]:
        """
        Returns (place_type, lifestyle_tags_dict).
        place_type is None if no mapping matches (skip this node).
        lifestyle_tags_dict example: {"vegan": True, "nature": True}
        """
        place_type = None
        for key, value, mapped_type in PLACE_TYPE_RULES:
            if key in osm_tags:
                if value == "*" or osm_tags[key] == value:
                    place_type = mapped_type
                    break

        if place_type is None:
            return None, {}

        lifestyle_tags = {}
        for tag_name, rules in LIFESTYLE_RULES.items():
            for osm_key, osm_val in rules:
                if osm_key in osm_tags:
                    if osm_val == "*" or osm_tags[osm_key] == osm_val:
                        lifestyle_tags[tag_name] = True
                        break

        return place_type, lifestyle_tags
    ```

    **backend/app/osm/parser.py:**
    Use pyosmium to parse PBF files and extract POI nodes:

    ```python
    import osmium

    class POIHandler(osmium.SimpleHandler):
        """Extract named POI nodes from PBF file."""
        def __init__(self):
            super().__init__()
            self.pois = []

        def node(self, n):
            tags = {t.k: t.v for t in n.tags}
            # Skip nodes without a name (unnamed POIs are not useful for discovery)
            if "name" not in tags:
                return
            place_type, lifestyle_tags = map_osm_tags(tags)
            if place_type is None:
                return
            self.pois.append({
                "osm_id": n.id,
                "name": tags["name"],
                "place_type": place_type,
                "lat": n.location.lat,
                "lon": n.location.lon,
                "tags": lifestyle_tags,
                "raw_osm_tags": tags,
                "description": tags.get("description", tags.get("wikipedia", None)),
            })

    def parse_pbf(file_path: str) -> list[dict]:
        """Parse a PBF file and return list of POI dicts."""
        handler = POIHandler()
        handler.apply_file(file_path, locations=True)
        return handler.pois
    ```

    **backend/app/osm/importer.py:**
    Bulk import parsed POIs into the database with upsert logic (ON CONFLICT osm_id DO UPDATE):

    ```python
    import asyncio
    import logging
    from sqlalchemy import text
    from sqlalchemy.dialects.postgresql import insert as pg_insert
    from app.db import async_session_maker
    from app.models.place import Place
    from app.osm.parser import parse_pbf

    logger = logging.getLogger(__name__)

    BATCH_SIZE = 1000

    async def import_pbf(file_path: str, region: str) -> dict:
        """
        Import a PBF file into the places table.
        Returns stats: {"total_parsed": N, "imported": M, "skipped": S}
        """
        logger.info(f"Parsing PBF file: {file_path}")
        pois = parse_pbf(file_path)
        logger.info(f"Parsed {len(pois)} POIs from {file_path}")

        imported = 0
        async with async_session_maker() as session:
            for i in range(0, len(pois), BATCH_SIZE):
                batch = pois[i:i + BATCH_SIZE]
                values = []
                for poi in batch:
                    values.append({
                        "osm_id": poi["osm_id"],
                        "name": poi["name"][:500],
                        "place_type": poi["place_type"],
                        "location": f"SRID=4326;POINT({poi['lon']} {poi['lat']})",
                        "tags": poi["tags"],
                        "description": poi.get("description"),
                        "source": "osm",
                        "raw_osm_tags": poi["raw_osm_tags"],
                        "region": region,
                    })
                stmt = pg_insert(Place).values(values)
                stmt = stmt.on_conflict_do_update(
                    index_elements=["osm_id"],
                    set_={
                        "name": stmt.excluded.name,
                        "place_type": stmt.excluded.place_type,
                        "location": stmt.excluded.location,
                        "tags": stmt.excluded.tags,
                        "description": stmt.excluded.description,
                        "raw_osm_tags": stmt.excluded.raw_osm_tags,
                        "region": stmt.excluded.region,
                    },
                )
                await session.execute(stmt)
                imported += len(batch)
                logger.info(f"Imported batch {i // BATCH_SIZE + 1}, total: {imported}")

            await session.commit()

        return {"total_parsed": len(pois), "imported": imported, "region": region}
    ```

    **backend/app/osm/cli.py:**
    CLI entry point using argparse (no extra dependencies):

    ```python
    import argparse
    import asyncio
    import logging
    import sys

    from app.osm.importer import import_pbf

    def main():
        parser = argparse.ArgumentParser(description="NomadBase OSM Import CLI")
        parser.add_argument("pbf_file", help="Path to the .osm.pbf file")
        parser.add_argument("--region", required=True, help="Region name (e.g., 'liechtenstein', 'austria')")
        parser.add_argument("--verbose", "-v", action="store_true", help="Verbose logging")

        args = parser.parse_args()

        logging.basicConfig(
            level=logging.DEBUG if args.verbose else logging.INFO,
            format="%(asctime)s [%(levelname)s] %(name)s: %(message)s",
        )

        result = asyncio.run(import_pbf(args.pbf_file, args.region))
        print(f"\nImport complete:")
        print(f"  Region: {result['region']}")
        print(f"  Parsed: {result['total_parsed']}")
        print(f"  Imported: {result['imported']}")

    if __name__ == "__main__":
        main()
    ```

    **backend/scripts/import_osm.sh:**
    Shell script that downloads a Geofabrik extract and runs the import:

    ```bash
    #!/bin/bash
    set -euo pipefail

    # Usage: ./import_osm.sh <region_path> <region_name>
    # Example: ./import_osm.sh europe/liechtenstein liechtenstein
    # Example: ./import_osm.sh europe/austria austria
    # Full list: https://download.geofabrik.de/

    REGION_PATH="${1:?Usage: $0 <geofabrik_region_path> <region_name>}"
    REGION_NAME="${2:?Usage: $0 <geofabrik_region_path> <region_name>}"
    DATA_DIR="/app/data/osm"
    GEOFABRIK_BASE="https://download.geofabrik.de"

    mkdir -p "$DATA_DIR"

    PBF_URL="${GEOFABRIK_BASE}/${REGION_PATH}-latest.osm.pbf"
    PBF_FILE="${DATA_DIR}/${REGION_NAME}-latest.osm.pbf"

    echo "=== NomadBase OSM Import ==="
    echo "Region: ${REGION_NAME}"
    echo "URL: ${PBF_URL}"
    echo "File: ${PBF_FILE}"
    echo ""

    # Download (with resume support)
    echo "Downloading ${REGION_NAME}..."
    wget -c -O "$PBF_FILE" "$PBF_URL"

    # Import
    echo "Importing ${REGION_NAME}..."
    python -m app.osm.cli "$PBF_FILE" --region "$REGION_NAME" -v

    echo ""
    echo "=== Import complete for ${REGION_NAME} ==="
    ```

    **backend/scripts/osm_cronjob.sh:**
    Cronjob wrapper script (designed to be called from cron or systemd timer):

    ```bash
    #!/bin/bash
    set -euo pipefail

    # Cronjob wrapper for OSM data updates
    # Add to crontab: 0 3 * * 0 docker compose exec api /app/scripts/osm_cronjob.sh
    # This runs every Sunday at 3 AM

    LOG_FILE="/app/data/osm/cronjob.log"
    REGIONS_FILE="/app/data/osm/regions.txt"

    echo "=== OSM Cronjob Started: $(date) ===" >> "$LOG_FILE"

    if [ ! -f "$REGIONS_FILE" ]; then
        echo "No regions.txt found at $REGIONS_FILE" >> "$LOG_FILE"
        echo "Create it with lines like: europe/liechtenstein liechtenstein" >> "$LOG_FILE"
        exit 0
    fi

    while IFS=' ' read -r region_path region_name; do
        [ -z "$region_path" ] && continue
        [[ "$region_path" =~ ^# ]] && continue
        echo "Processing: $region_name" >> "$LOG_FILE"
        /app/scripts/import_osm.sh "$region_path" "$region_name" >> "$LOG_FILE" 2>&1 || echo "FAILED: $region_name" >> "$LOG_FILE"
    done < "$REGIONS_FILE"

    echo "=== OSM Cronjob Finished: $(date) ===" >> "$LOG_FILE"
    ```
